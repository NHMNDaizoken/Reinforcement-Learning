### Task 2: Generator cấu hình CityFlow

**File:** `configs/generate_configs.py`

**Tại sao cần script sinh tự động thay vì viết tay JSON:**
CityFlow yêu cầu `roadnet.json` mô tả *chính xác* ID của từng nút giao, từng đoạn đường và lane. Viết tay cho lưới 3×3 (9 nút, 24 đoạn đường, 72+ lane) rất dễ sai. Script đảm bảo tính nhất quán giữa topology và flow.

**Tại sao dùng phân phối Poisson cho flow:**
Giao thông thực tế có tính ngẫu nhiên — xe không đến đều đặn theo giây. Poisson (interval = 3600 / vph) mô phỏng đúng quá trình đến ngẫu nhiên độc lập, cho dataset huấn luyện đa dạng hơn.

- [ ] Viết `configs/generate_configs.py`

```python
# configs/generate_configs.py
"""
Sinh roadnet.json (lưới NxN) và flow_*.json (3 mức lưu lượng).

Quy ước ID:
  - Nút: node_{row}_{col}
  - Đường: road_{src}_{dst}  (từng chiều một chiều)
  - Lane:  road_{src}_{dst}_{lane_idx}
"""
import json, itertools
from pathlib import Path

GRID_N      = 3
ROAD_LEN    = 300      # mét — đủ dài để xe không tràn ra ngoài lưới
MAX_SPEED   = 11.11    # m/s ≈ 40 km/h
LANE_WIDTH  = 3.5
LANES       = 3        # số làn mỗi chiều
OUT_DIR     = Path(__file__).parent


def node(r, c):    return f"node_{r}_{c}"
def road(a, b):    return f"road_{a}_{b}"
def lane(a, b, k): return f"road_{a}_{b}_{k}"


def build_roadnet(n: int = GRID_N) -> dict:
    nodes, roads = [], []
    node_roads: dict[str, list[str]] = {node(r, c): [] for r, c in itertools.product(range(n), range(n))}

    def add_road(src, dst):
        rid = road(src, dst)
        roads.append({
            "id": rid,
            "startIntersection": src,
            "endIntersection": dst,
            "points": [],
            "lanes": [{"width": LANE_WIDTH, "maxSpeed": MAX_SPEED} for _ in range(LANES)],
        })
        node_roads[src].append(rid)
        node_roads[dst].append(rid)

    for r, c in itertools.product(range(n), range(n)):
        if c + 1 < n:
            add_road(node(r, c), node(r, c + 1))
            add_road(node(r, c + 1), node(r, c))
        if r + 1 < n:
            add_road(node(r, c), node(r + 1, c))
            add_road(node(r + 1, c), node(r, c))

    for r, c in itertools.product(range(n), range(n)):
        nid = node(r, c)
        # CityFlow cần trafficLight.lightphases để nhận set_tl_phase
        # Mỗi phase là danh sách road link index — để trống để RL tự quyết
        nodes.append({
            "id": nid,
            "point": {"x": c * ROAD_LEN, "y": r * ROAD_LEN},
            "width": 10,
            "roads": node_roads[nid],
            "roadLinks": [],
            "trafficLight": {"roadLinkIndices": [], "lightphases": [
                {"time": 30, "availableRoadLinks": []},  # phase 0
                {"time": 30, "availableRoadLinks": []},  # phase 1
            ]},
            "virtual": False,
        })

    return {"intersections": nodes, "roads": roads}


def build_flow(vph: int, seed: int = 42) -> list:
    """
    Tạo 4 tuyến đường chéo qua lưới, mỗi tuyến 1 luồng xe.
    interval = 3600/vph mô phỏng Poisson arrival với rate = vph/3600 xe/giây.
    """
    import random; random.seed(seed)
    n = GRID_N
    interval = 3600 / vph
    base_vehicle = {
        "length": 5.0, "width": 2.0,
        "maxPosAcc": 2.0, "maxNegAcc": 4.5,
        "usualPosAcc": 2.0, "usualNegAcc": 4.5,
        "minGap": 2.5, "maxSpeed": MAX_SPEED, "headwayTime": 2.0,
    }
    # 4 tuyến: ngang trái→phải, ngang phải→trái, dọc trên→dưới, dọc dưới→trên
    routes = [
        [road(node(r, c), node(r, c + 1)) for c in range(n - 1)]  # hàng r, trái→phải
        for r in range(n)
    ] + [
        [road(node(r, c + 1), node(r, c)) for c in range(n - 2, -1, -1)]  # hàng r, phải→trái
        for r in range(n)
    ]
    # Chỉ giữ 4 tuyến đại diện để tránh flow quá dày
    routes = routes[:4]
    return [{"vehicle": base_vehicle, "route": route,
             "interval": interval, "startTime": 0, "endTime": 3600}
            for route in routes]


if __name__ == "__main__":
    roadnet = build_roadnet()
    (OUT_DIR / "roadnet.json").write_text(json.dumps(roadnet, indent=2))
    print("roadnet.json ✓")

    for label, vph in [("low", 300), ("medium", 600), ("high", 900)]:
        flow = build_flow(vph)
        (OUT_DIR / f"flow_{label}.json").write_text(json.dumps(flow, indent=2))
        print(f"flow_{label}.json ✓  ({vph} veh/h, interval={3600/vph:.1f}s)")
```

### Task 3: MaxPressure Baseline

**File:** `src/baseline.py`, `tests/test_baseline.py`

**Tại sao Baseline này quan trọng:**
MaxPressure là thuật toán greedy tối ưu cục bộ — nó chọn pha có áp lực cao nhất tại mỗi bước. Theo lý thuyết dòng mạng, nó tối đa hóa throughput trong điều kiện stationary. DQN phải *vượt* baseline này nhờ discount factor γ cho phép tính toán dài hạn (tránh chuyển điểm ùn tắc sang ngã tư kế tiếp).

**Công thức:** `pressure_phase = Σ (q_in - q_out)` cho tất cả movement trong phase đó. Chọn `argmax`.

```python
# src/baseline.py
"""
MaxPressure rule-based controller — không dùng học máy.

Đây là lower bound để so sánh: greedy tối ưu cục bộ, không có tầm nhìn dài hạn.
DQN phải vượt controller này sau khi huấn luyện.
"""
import cityflow


class MaxPressureBaseline:
    def __init__(self, engine: cityflow.Engine,
                 phase_map: dict[str, list[list[tuple[str, str]]]]):
        """
        phase_map: {inter_id: [[(in_lane, out_lane), ...], ...]}
          - Chiều ngoài: danh sách các phase
          - Chiều trong: danh sách (in_lane_id, out_lane_id) trong phase đó
        """
        self.engine = engine
        self.phase_map = phase_map

    def select_actions(self) -> dict[str, int]:
        counts = self.engine.get_lane_vehicle_count()
        return {
            inter_id: max(
                range(len(phases)),
                key=lambda k: sum(
                    counts.get(in_l, 0) - counts.get(out_l, 0)
                    for in_l, out_l in phases[k]
                )
            )
            for inter_id, phases in self.phase_map.items()
        }
```

### Task 4: TrafficEnv

**File:** `src/env.py`, `tests/test_env.py`

**Tại sao state bao gồm cả `q_out`:**
Nếu chỉ quan sát `q_in`, tác tử không biết đường thoát phía trước có kẹt không — nó có thể bật đèn xanh rồi gây gridlock. `q_out` là thông tin Markov cần thiết để quyết định có an toàn để giải phóng xe không (đúng như điều kiện `P ≤ 0` trong MaxPressure).

**Tại sao one-hot phase trong state:**
Không có thông tin pha hiện tại, tác tử không biết nó đang ở trạng thái nào — cùng một `[q_in, q_out]` nhưng ở pha khác nhau cần quyết định khác nhau. One-hot đảm bảo tính Markov của state.

**Tại sao `sim_steps_per_action = 10`:**
CityFlow chạy theo bước 1 giây. Một quyết định đèn giao thông thực tế kéo dài ~10 giây. Gộp 10 bước sim cho 1 action step giảm tần suất quyết định xuống mức thực tế và giảm kích thước bộ nhớ replay buffer.

```python
# src/env.py
"""
TrafficEnv — wrapper CityFlow cho RL.

State:  s_i = [q_in_1..k, q_out_1..k, one_hot(phase)]
Action: chỉ số phase (integer)
Reward: r_i = -(Σq_in - Σq_out)  [âm MaxPressure]

Thiết kế local observation: mỗi agent chỉ nhìn thấy dữ liệu của giao lộ mình.
Điều này cho phép Parameter Sharing hoạt động — tất cả agent nhận state cùng kích thước
và học cùng một policy, bất kể vị trí trong lưới.
"""
import json, tempfile, os
import numpy as np
import cityflow


class TrafficEnv:
    def __init__(self, roadnet_path: str, flow_path: str,
                 phase_map: dict[str, list[list[tuple[str, str]]]],
                 sim_steps_per_action: int = 10):
        """
        sim_steps_per_action: số bước sim CityFlow (1s/bước) cho mỗi quyết định RL.
        Giá trị 10 tương ứng ~10 giây thực — phù hợp tần suất đổi đèn thực tế.
        """
        self.phase_map = phase_map
        self.inter_ids = list(phase_map.keys())
        self.n_agents = len(self.inter_ids)
        self.sim_steps_per_action = sim_steps_per_action
        self._roadnet = roadnet_path
        self._flow = flow_path
        self.engine: cityflow.Engine | None = None
        self._current_phases: dict[str, int] = {}

    def reset(self) -> list[np.ndarray]:
        if self.engine is not None:
            del self.engine
        cfg = {
            "interval": 1.0, "seed": 0, "dir": "./",
            "roadnetFile": self._roadnet, "flowFile": self._flow,
            "rlTrafficLight": True, "saveReplay": False,
            "roadnetLogFile": "", "replayLogFile": "",
            "laneChange": False, "threadNum": 1,
        }
        path = os.path.join(tempfile.gettempdir(), "cf_cfg.json")
        with open(path, "w") as f:
            json.dump(cfg, f)
        self.engine = cityflow.Engine(path, thread_num=1)
        self._current_phases = {iid: 0 for iid in self.inter_ids}
        for iid in self.inter_ids:
            self.engine.set_tl_phase(iid, 0)
        for _ in range(self.sim_steps_per_action):
            self.engine.next_step()
        return self._get_states()

    def step(self, actions: dict[str, int]):
        """actions: {inter_id: phase_index}"""
        for iid, phase in actions.items():
            self.engine.set_tl_phase(iid, phase)
            self._current_phases[iid] = phase
        for _ in range(self.sim_steps_per_action):
            self.engine.next_step()
        states  = self._get_states()
        rewards = self._get_rewards()
        done    = self.engine.get_current_time() >= 3600
        info    = {
            "atl":        self.engine.get_average_travel_time(),
            "throughput": self.engine.get_vehicle_count(),
        }
        return states, rewards, done, info

    # ------------------------------------------------------------------ #
    def _get_states(self) -> list[np.ndarray]:
        counts = self.engine.get_lane_vehicle_count()
        states = []
        for iid in self.inter_ids:
            phases = self.phase_map[iid]
            q_in, q_out = [], []
            for movements in phases:
                for in_l, out_l in movements:
                    q_in.append(float(counts.get(in_l, 0)))
                    q_out.append(float(counts.get(out_l, 0)))
            one_hot = np.zeros(len(phases), dtype=np.float32)
            one_hot[self._current_phases[iid]] = 1.0
            states.append(np.concatenate([q_in, q_out, one_hot]).astype(np.float32))
        return states

    def _get_rewards(self) -> list[float]:
        counts = self.engine.get_lane_vehicle_count()
        rewards = []
        for iid in self.inter_ids:
            pressure = sum(
                counts.get(in_l, 0) - counts.get(out_l, 0)
                for movements in self.phase_map[iid]
                for in_l, out_l in movements
            )
            rewards.append(-float(pressure))
        return rewards

    @property
    def state_dim(self) -> int:
        phases = self.phase_map[self.inter_ids[0]]
        n_movements = sum(len(mvs) for mvs in phases)
        return 2 * n_movements + len(phases)

    @property
    def action_dim(self) -> int:
        return len(self.phase_map[self.inter_ids[0]])
```

## Phase 2 — DQN Core & Training

### Task 5: SharedDQNAgent

**File:** `src/agent.py`, `tests/test_agent.py`

**Tại sao Parameter Sharing:**
Thay vì N mạng riêng ($N \times |\theta|$ tham số), tất cả 9 agent dùng chung 1 mạng ($|\theta|$ tham số). Tại mỗi bước sim, 9 agent tạo ra 9 transition → sample efficiency ×9. Hội tụ nhanh hơn nhiều trong thời gian 14 ngày.

**Tại sao Huber Loss thay vì MSE:**
Khi xe kẹt nặng, reward $r$ biến động lớn → TD error lớn → MSE cho gradient rất lớn → mô hình phân kỳ. Huber Loss = MSE khi error nhỏ, MAE khi error lớn → gradient bị clip tự nhiên ở vùng sai số lớn.

**Tại sao Target Network:**
Nếu chỉ dùng 1 mạng, target $y_j = r + \gamma \max_a Q(s', a; \theta)$ và prediction $Q(s, a; \theta)$ cùng thay đổi → mô hình "rượt đuổi chính nó". Target network ($\theta^-$) được cập nhật chậm (mỗi C=200 steps) → target ổn định → huấn luyện hội tụ.

**Tại sao ε-greedy:**
Giai đoạn đầu cần khám phá: bật ngẫu nhiên đèn ở các pha khác nhau để thu thập đa dạng kinh nghiệm. Nếu chỉ khai thác từ đầu, mạng sẽ bị stuck tại local policy tệ. ε giảm dần từ 1.0 → 0.01 theo exponential decay khi đã đủ kinh nghiệm.

```python
# src/agent.py
"""
SharedDQNAgent — một mạng DQN cho toàn bộ N agent (Parameter Sharing).

Hyperparameters được chọn cho sprint 14 ngày:
  buffer_size=5000, batch=64: train nhanh, đủ đa dạng dữ liệu.
  target_update_freq=200: neo giữ target ổn định mà không cập nhật quá chậm.
  gamma=0.95: tầm nhìn ~20 bước — đủ để học "không đẩy ùn tắc sang ngã tư kế".
"""
import random
from collections import deque

import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim


class QNetwork(nn.Module):
    """MLP 128→64 — đủ sâu để học policy phi tuyến, không overfit với dataset nhỏ."""
    def __init__(self, state_dim: int, action_dim: int):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(state_dim, 128), nn.ReLU(),
            nn.Linear(128, 64),        nn.ReLU(),
            nn.Linear(64, action_dim),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.net(x)


class ReplayBuffer:
    def __init__(self, capacity: int):
        self._buf: deque = deque(maxlen=capacity)

    def push(self, s, a, r, ns, done):
        self._buf.append((s, a, r, ns, done))

    def sample(self, batch_size: int):
        s, a, r, ns, d = zip(*random.sample(self._buf, batch_size))
        return (np.stack(s), np.array(a), np.array(r, dtype=np.float32),
                np.stack(ns), np.array(d, dtype=np.float32))

    def __len__(self): return len(self._buf)


class SharedDQNAgent:
    def __init__(self, state_dim: int, action_dim: int, n_agents: int,
                 lr: float = 1e-3, gamma: float = 0.95,
                 epsilon_start: float = 1.0, epsilon_end: float = 0.01,
                 epsilon_decay: float = 0.995,
                 buffer_size: int = 5_000, batch_size: int = 64,
                 target_update_freq: int = 200, huber_delta: float = 1.0,
                 device: str = "cpu"):
        self.action_dim = action_dim
        self.gamma = gamma
        self.epsilon = epsilon_start
        self.epsilon_end = epsilon_end
        self.epsilon_decay = epsilon_decay
        self.batch_size = batch_size
        self.target_update_freq = target_update_freq
        self.device = torch.device(device)

        self.policy_net = QNetwork(state_dim, action_dim).to(self.device)
        self.target_net = QNetwork(state_dim, action_dim).to(self.device)
        self.target_net.load_state_dict(self.policy_net.state_dict())
        self.target_net.eval()

        self.optimizer = optim.Adam(self.policy_net.parameters(), lr=lr)
        self.loss_fn   = nn.HuberLoss(delta=huber_delta)
        self.buffer    = ReplayBuffer(buffer_size)
        self._steps    = 0

    def select_actions(self, states: list[np.ndarray]) -> list[int]:
        return [
            random.randrange(self.action_dim) if random.random() < self.epsilon
            else int(self.policy_net(
                torch.FloatTensor(s).unsqueeze(0).to(self.device)
            ).argmax(1).item())
            for s in states
        ]

    def push_transitions(self, states, actions, rewards, next_states, dones):
        for s, a, r, ns, d in zip(states, actions, rewards, next_states, dones):
            self.buffer.push(s, a, r, ns, d)

    def learn(self) -> float | None:
        if len(self.buffer) < self.batch_size:
            return None
        s, a, r, ns, d = self.buffer.sample(self.batch_size)
        s  = torch.FloatTensor(s).to(self.device)
        a  = torch.LongTensor(a).to(self.device)
        r  = torch.FloatTensor(r).to(self.device)
        ns = torch.FloatTensor(ns).to(self.device)
        d  = torch.FloatTensor(d).to(self.device)

        q_pred = self.policy_net(s).gather(1, a.unsqueeze(1)).squeeze(1)
        with torch.no_grad():
            q_target = r + self.gamma * self.target_net(ns).max(1).values * (1 - d)

        loss = self.loss_fn(q_pred, q_target)
        self.optimizer.zero_grad()
        loss.backward()
        self.optimizer.step()

        self._steps += 1
        if self._steps % self.target_update_freq == 0:
            self.target_net.load_state_dict(self.policy_net.state_dict())

        self.epsilon = max(self.epsilon_end, self.epsilon * self.epsilon_decay)
        return float(loss.item())

    def save(self, path: str): torch.save(self.policy_net.state_dict(), path)

    def load(self, path: str):
        self.policy_net.load_state_dict(torch.load(path, map_location=self.device))
        self.target_net.load_state_dict(self.policy_net.state_dict())
```

ATL (Average Travel Time) là metric thực tế cuối cùng mà người dùng quan tâm. Reward âm MaxPressure là proxy — có thể tăng nhưng ATL lại tệ hơn trong edge case. Chọn checkpoint theo ATL đảm bảo model tốt nhất theo tiêu chí thực tế.

```python
# src/train.py
"""
Training loop — 500 episodes, shared DQN, Parameter Sharing.

Chạy: python src/train.py [--roadnet ...] [--flow ...] [--episodes N]
Output:
  - data/buffer_dqn.csv: toàn bộ transitions để build offline dataset
  - models/best.pth: weights tốt nhất theo ATL
"""
import csv, os, json, argparse
import numpy as np
from src.env import TrafficEnv
from src.agent import SharedDQNAgent


def build_phase_map(roadnet_path: str) -> dict[str, list[list[tuple[str, str]]]]:
    """
    Trích xuất phase_map từ roadnet.json.
    Với lưới 3×3, mỗi ngã tư có 4 đường nối → chia thành 2 phase:
      phase 0: horizontal (east-west)
      phase 1: vertical   (north-south)
    Cách chia: sort road IDs → nửa đầu phase 0, nửa sau phase 1.
    """
    with open(roadnet_path) as f:
        roadnet = json.load(f)
    phase_map = {}
    for inter in roadnet["intersections"]:
        if inter.get("virtual", False):
            continue
        roads = sorted(inter["roads"])
        half  = len(roads) // 2 or 1
        def to_lane(road_id): return f"{road_id}_0"
        # Mỗi phase: list of (in_lane, out_lane) từ đường vào → đường ra đối diện
        phase_0 = [(to_lane(roads[i]), to_lane(roads[i + half])) for i in range(half)]
        phase_1 = [(to_lane(roads[i]), to_lane(roads[i - half])) for i in range(half, len(roads))]
        phase_map[inter["id"]] = [phase_0, phase_1 or phase_0]
    return phase_map


def train(roadnet: str, flow: str, episodes: int = 500,
          out_csv: str = "data/buffer_dqn.csv",
          model_path: str = "models/best.pth"):
    phase_map = build_phase_map(roadnet)
    env   = TrafficEnv(roadnet, flow, phase_map)
    agent = SharedDQNAgent(state_dim=env.state_dim, action_dim=env.action_dim,
                           n_agents=env.n_agents)
    os.makedirs(os.path.dirname(out_csv),    exist_ok=True)
    os.makedirs(os.path.dirname(model_path), exist_ok=True)
    best_atl = float("inf")

    with open(out_csv, "w", newline="") as f:
        w = csv.writer(f)
        w.writerow(["episode", "step", "agent_id", "state_vec", "action",
                    "reward", "next_state_vec", "done", "atl", "throughput"])

        for ep in range(episodes):
            states = env.reset()
            done, step, ep_reward = False, 0, 0.0

            while not done:
                actions = agent.select_actions(states)
                action_map = {env.inter_ids[i]: actions[i] for i in range(env.n_agents)}
                next_states, rewards, done, info = env.step(action_map)
                agent.push_transitions(states, actions, rewards, next_states,
                                       [done] * env.n_agents)
                loss = agent.learn()
                ep_reward += sum(rewards)

                for i, iid in enumerate(env.inter_ids):
                    w.writerow([ep, step, iid,
                                 states[i].tolist(), actions[i], rewards[i],
                                 next_states[i].tolist(), done,
                                 info["atl"], info["throughput"]])
                states = next_states
                step  += 1

            atl = info["atl"]
            loss_str = f"{loss:.4f}" if loss is not None else "N/A"
            print(f"EP {ep:4d} | reward={ep_reward:8.1f} | atl={atl:.2f}s "
                  f"| ε={agent.epsilon:.3f} | loss={loss_str}")

            if atl < best_atl:
                best_atl = atl
                agent.save(model_path)
                print(f"         ↳ best ATL={best_atl:.2f}s → {model_path}")


if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--roadnet",  default="configs/roadnet.json")
    p.add_argument("--flow",     default="configs/flow_medium.json")
    p.add_argument("--episodes", type=int, default=500)
    args = p.parse_args()
    train(args.roadnet, args.flow, args.episodes)
```

```python
def run_baseline(roadnet: str, flow: str, episodes: int = 50,
                 out_csv: str = "data/buffer_baseline.csv"):
    phase_map = build_phase_map(roadnet)
    env = TrafficEnv(roadnet, flow, phase_map)
    os.makedirs(os.path.dirname(out_csv), exist_ok=True)

    with open(out_csv, "w", newline="") as f:

                for i, iid in enumerate(env.inter_ids):

                                 states[i].tolist(), actions[i], rewards[i],
                                 next_states[i].tolist(), done,
                                 info["atl"], info["throughput"]])
                states = next_states
                step  += 1
            print(f"Baseline EP {ep:3d} | atl={info['atl']:.2f}s")


def merge_to_sqlite(csv_files: list[str], db_path: str = "data/offline_dataset.db"):
    os.makedirs(os.path.dirname(db_path), exist_ok=True)
    conn = sqlite3.connect(db_path)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS transitions (
            id           INTEGER PRIMARY KEY AUTOINCREMENT,
            source       TEXT,
            episode      INTEGER,
            step         INTEGER,
            agent_id     TEXT,
            state_vec    TEXT,
            action       INTEGER,
            reward       REAL,
            next_state_vec TEXT,
            done         INTEGER,
            atl          REAL,
            throughput   INTEGER
        )
    """)
    conn.execute("CREATE INDEX IF NOT EXISTS idx_src_ep ON transitions(source, episode)")
    for csv_path in csv_files:
        source = os.path.basename(csv_path).replace(".csv", "")
        with open(csv_path) as f:
            rows = [
                (source, int(r["episode"]), int(r["step"]), r["agent_id"],
                 r["state_vec"], int(r["action"]), float(r["reward"]),
                 r["next_state_vec"], int(r["done"] == "True"),
                 float(r["atl"]), int(float(r["throughput"])))
                for r in __import__("csv").DictReader(f)
            ]
        conn.executemany(
            "INSERT INTO transitions "
            "(source,episode,step,agent_id,state_vec,action,reward,"
            "next_state_vec,done,atl,throughput) VALUES (?,?,?,?,?,?,?,?,?,?,?)",
            rows
        )
    conn.commit(); conn.close()
    print(f"Merged {csv_files} → {db_path}")


if __name__ == "__main__":
    p = argparse.ArgumentParser()
    p.add_argument("--roadnet",  default="configs/roadnet.json")
    p.add_argument("--flow",     default="configs/flow_medium.json")
```

```python
import sqlite3, json
from dataclasses import dataclass

DB_PATH = "data/offline_dataset.db"


@dataclass
class StepRecord:
    step: int; agent_id: str; state_vec: list[float]
    action: int; reward: float; atl: float; throughput: int


def get_episodes(source: str = "buffer_dqn") -> list[int]:
    conn = sqlite3.connect(DB_PATH)
    rows = conn.execute(
        "SELECT DISTINCT episode FROM transitions WHERE source=? ORDER BY episode",
        (source,)
    ).fetchall()
    conn.close()
    return [r[0] for r in rows]


def get_replay(episode: int, source: str = "buffer_dqn") -> list[StepRecord]:
    conn = sqlite3.connect(DB_PATH)
    rows = conn.execute(
        "SELECT step, agent_id, state_vec, action, reward, atl, throughput "
        "FROM transitions WHERE source=? AND episode=? ORDER BY step, agent_id",
        (source, episode)
    ).fetchall()
    conn.close()
    return [StepRecord(r[0], r[1], json.loads(r[2]), r[3], r[4], r[5], r[6]) for r in rows]
```

```tsx
// web/components/GridCanvas.tsx
"use client";
import { useEffect, useRef } from "react";

const GRID = 3, CELL = 120;
// Màu tương ứng với chỉ số phase: xanh lá = đang bật xanh, đỏ = dừng
const PHASE_COLORS = ["#22c55e", "#ef4444", "#f59e0b", "#3b82f6"];

interface StepData { step: number; agent_id: string; action: number }
interface Props { steps: StepData[]; currentStep: number }

export default function GridCanvas({ steps, currentStep }: Props) {
  const ref = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const ctx = ref.current?.getContext("2d");
    if (!ctx) return;
    ctx.clearRect(0, 0, GRID * CELL, GRID * CELL);

    // Vẽ đường
    ctx.strokeStyle = "#374151"; ctx.lineWidth = 2;
    for (let r = 0; r < GRID; r++) {
      for (let c = 0; c < GRID; c++) {
        const [x, y] = [c * CELL + CELL / 2, r * CELL + CELL / 2];
        if (c + 1 < GRID) { ctx.beginPath(); ctx.moveTo(x, y); ctx.lineTo(x + CELL, y); ctx.stroke(); }
        if (r + 1 < GRID) { ctx.beginPath(); ctx.moveTo(x, y); ctx.lineTo(x, y + CELL); ctx.stroke(); }
      }
    }

    // Vẽ ngã tư với màu phase hiện tại
    const cur = steps.filter(s => s.step === currentStep);
    for (let r = 0; r < GRID; r++) {
      for (let c = 0; c < GRID; c++) {
        const agent = cur.find(s => s.agent_id === `node_${r}_${c}`);
        const [x, y] = [c * CELL + CELL / 2, r * CELL + CELL / 2];
        ctx.beginPath(); ctx.arc(x, y, 14, 0, 2 * Math.PI);
        ctx.fillStyle = agent ? PHASE_COLORS[agent.action % PHASE_COLORS.length] : "#9ca3af";
        ctx.fill(); ctx.strokeStyle = "#111"; ctx.lineWidth = 1.5; ctx.stroke();
        ctx.fillStyle = "#fff"; ctx.font = "10px sans-serif";
        ctx.fillText(`${r},${c}`, x - 8, y + 4);
      }
    }
  }, [steps, currentStep]);

  return <canvas ref={ref} width={GRID * CELL} height={GRID * CELL} className="border rounded" />;
}
```

```tsx
// web/app/page.tsx
"use client";
import { useEffect, useState, useRef } from "react";
import GridCanvas from "@/components/GridCanvas";
import Charts from "@/components/Charts";

const API = "http://localhost:8000";

export default function Home() {
  const [episodes, setEpisodes] = useState<number[]>([]);
  const [episode,  setEpisode]  = useState(0);
  const [steps,    setSteps]    = useState<any[]>([]);
  const [curStep,  setCurStep]  = useState(0);
  const [playing,  setPlaying]  = useState(false);
  const timer = useRef<ReturnType<typeof setInterval> | null>(null);

  useEffect(() => {
    fetch(`${API}/episodes`).then(r => r.json()).then((eps: number[]) => {
      setEpisodes(eps);
      if (eps.length) setEpisode(eps[0]);
    });
  }, []);

  useEffect(() => {
    if (!episodes.length) return;
    fetch(`${API}/replay?episode=${episode}`)
      .then(r => r.json()).then((data: any[]) => { setSteps(data); setCurStep(0); });
  }, [episode]);

  useEffect(() => {
    if (timer.current) clearInterval(timer.current);
    if (!playing) return;
    timer.current = setInterval(() => {
      setCurStep(cs => {
        const max = steps.length ? Math.max(...steps.map((s: any) => s.step)) : 0;
        if (cs >= max) { setPlaying(false); return cs; }
        return cs + 1;
      });
    }, 150);
    return () => { if (timer.current) clearInterval(timer.current); };
  }, [playing, steps]);

  const atlData = Array.from(
    steps.reduce((m: Map<number, number>, s: any) => m.set(s.step, s.atl), new Map()),
    ([step, atl]) => ({ step, atl })
  ).sort((a, b) => a.step - b.step);

  return (
    <main className="p-6 space-y-4">
      <h1 className="text-2xl font-bold">Traffic Signal Digital Twin</h1>
      <div className="flex gap-4 items-center">
        <label className="text-sm">Episode:
          <select className="ml-2 border rounded px-2 py-1" value={episode}
            onChange={e => setEpisode(+e.target.value)}>
            {episodes.map(ep => <option key={ep}>{ep}</option>)}
          </select>
        </label>
        <button onClick={() => setPlaying(p => !p)}
          className="px-4 py-1 bg-blue-600 text-white rounded text-sm">
          {playing ? "Pause" : "Play"}
        </button>
        <span className="text-sm text-gray-500">Step: {curStep}</span>
      </div>
      <div className="flex gap-6">
        <GridCanvas steps={steps} currentStep={curStep} />
        <Charts data={atlData} />
      </div>
    </main>
  );
}
```
